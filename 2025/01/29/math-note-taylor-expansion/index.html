Expansiones de Taylor

\begin{align*} \frac{1}{1 - x} &= \sum_{n=0}^{\infty} x^n & x \in (-1, 1) \\\\ e^x &= \sum_{n=0}^{\infty} \frac{x^n}{n!} & x \in \mathbb{R} \\\\ \ln(1 + x) &= \sum_{n=0}^{\infty} \frac{(-1)^n x^{n+1}}{n+1} & x \in (-1, 1] \\\\ \sin x &= \sum_{n=0}^{\infty} \frac{(-1)^n x^{2n+1}}{(2n+1)!} & x \in \mathbb{R} \\\\ \cos x &= \sum_{n=0}^{\infty} \frac{(-1)^n x^{2n}}{(2n)!} & x \in \mathbb{R} \\\\ \arctan x &= \sum_{n=0}^{\infty} \frac{(-1)^n x^{2n+1}}{2n+1} & x \in [-1, 1] \\\\ (1 + x)^r &= \sum_{n=0}^{\infty} \binom{r}{n} x^n & x \in (-1, 1) \end{align*}
